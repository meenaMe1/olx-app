select u.USERID,u.USERNAME,up.GRADE,s.PRACTICE_NAME,s.SUB_PRACTICE_NAME,s.SKILL_FAMILY_NAME,s.SKILL_NAME,u.REQUESTDATE,u.ACTIONBY,p.EMPLOYEE_NAME,p.GRADE as 'managerGrade', u.ACTIONDATE,u.REQUESTSTATUS from APPS.XXG4_COMPETENCY_SKILL_V s left join ZEN.XXG4_HRMS_USERSKILLCHREQ_TBL c on c.SKILLID=s.SKILL_ID 
left join ZEN.XXG4_HRMS_USERCHANGEREQ_TBL u on u.USERCHANGEREQUESTID=c.USERCHANGEREQUESTID left join ZEN.XXG4_ZENHELP_ASSO_USRPROFL_TBL up on up.EMPLOYEE_NUMBER=u.USERID 
left join ZEN.XXG4_ZENHELP_ASSO_USRPROFL_TBL p on p.EMPLOYEE_NUMBER=u.ACTIONBY where u.USERID is not null order by u.REQUESTDATE desc
